onchnage

监听select的change事件，获取当前选择的item 

var change = new UISelect({
	children:country
});
change.on('change',function(item){
	alert(JSON.stringify(item));
	this.hide();
});
$('#select-change').click(function() {
	change.show();
});
show code

value

获取select当前选中的item 

var value = new UISelect({
	children:country
});					
$('#select-value').click(function() {
	value.show();
});
$('#get-value').click(function(){
	alert(JSON.stringify(value.value()));
	value.hide();
});

selected

var data = _.extend({},country);
data[3].selected = true;

Options

通过设置options参数，改变select的显示方式 

var options = new UISelect({
	//是否开启动画效果
	fade: false,
	//是否有半透明遮罩
	backdrop: false,
	//遮罩是否响应点击关闭事件
	keybord: false,
	//数据
	children:country
});
options.on('change',function(item){
	alert(JSON.stringify(item));
	this.hide();
})
$('#select-options').click(function() {
	options.show();
});

Event

监听select的事件 

var evt = new UISelect({
	children:country
});
evt.on('shown',function(){
	alert('shown');
});
evt.on('hidden',function(){
	alert('hidden');
});
$('#select-event').click(function(){
	evt.show();
});
