Confirm

confirm确认框，带半透明遮罩、动画，响应keybord关闭 

var normal = new UIConfirm();					
$('#confirm').click(function() {
	normal.show("这是Confirm的基本用法");
});
show code

Title

设置confirm的标题栏 

var title = new UIConfirm();					
$('#confirm-title').click(function() {
	title.show("这是Confirm的基本用法",'Confirm Title');
});
show code

Callback

设置confirm的“确定”和“取消”的回调函数，取消函数默认为关闭confirm，注意 如果有动画效果，则回调函数会在动画执行之前就被触发，如果需要监听动画完全执行完毕的事件，请参见 event 

var callabck = new UIConfirm();					
$('#confirm-callabck').click(function() {
	callabck.show("设置回调函数",function(){
		alert('OK')
	},function(){
		alert('Cancel');
		this.hide();
	});
});
show code

Options

通过设置options参数，改变toast的显示方式 

var options = new UIConfirm({
	//是否开启动画效果
	fade: false,
	//是否有半透明遮罩
	backdrop: false,
	//遮罩是否响应点击关闭事件
	keybord: false
});
$('#confirm-options').click(function() {
	options.show("使用了options参数的confirm",'Confirm Title',function(){
		alert('OK');
	},function(){
		alert('Cancel');
		this.hide();
	});
});
show code

Event

监听confirm的事件 

var evt = new UIConfirm();
evt.on('shown',function(){
	alert('shown');
});
evt.on('hidden',function(){
	alert('hidden');
});
$('#confirm-event').click(function(){
	evt.show('事件监听confirm',function(){
		alert('OK');
	},function(){
		alert('Cancel');
		this.hide();
	})
});
