Toast

自动关闭信息提示框基本用法，带半透明遮罩，动画，2秒后自动关闭 

var normal = new UIToast();					
$('#toast').click(function() {
	normal.show("这是UIToast的基本用法");
});

Duration

设置延时关闭时间，“秒”为单位 

var duration = new UIToast();					
$('#toast-duration').click(function() {
	duration.show("3.5秒后我将自动关闭",3.5);
});

Callback

设置toast自动关闭后的回调函数，注意 如果有动画效果，则回调函数会在动画执行之前就被触发，如果需要监听动画完全执行完毕的事件，请参见 event 

var callback = new UIToast();					
$('#toast-duration-callback').click(function() {
	callback.show("3.5秒后我将自动关闭，并触发回调函数",3.5,function(){
		alert('duration callback');
	});
});
$('#toast-callback').click(function() {
	callback.show("有回调函数的toast",function(){
		alert('callback');
	});
});

Options

通过设置options参数，改变toast的显示方式 

var options = new UIToast({
	//是否开启动画效果
	fade: false,
	//是否有半透明遮罩
	backdrop: false,
	//遮罩是否响应点击关闭事件
	keybord: false
});
$('#toast-options').click(function() {
	options.show("改变了显示方式的toast，并于3.5秒后将自动关闭",3.5);
});

Event

监听toast的事件 

var evt = new UIToast();
evt.on('shown',function(){
	alert('shown');
});
evt.on('hidden',function(){
	alert('hidden');
});
$('#toast-event').click(function(){
	evt.show('事件监听toast',function(){
		alert('callback hidden');
	})
});
