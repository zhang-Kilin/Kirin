onselect

监听list的 select 事件，获取当前选择的item 

var change = new UIList({
	children:country
});
change.on('select',function(item){
	alert(JSON.stringify(item));
});
$('#list-select').click(function() {
	change.show();
});

Options

通过设置options参数，改变list的显示方式 

var options = new UIList({
	//是否开启动画效果
	fade: false,
	//是否有半透明遮罩
	backdrop: false,
	//遮罩是否响应点击关闭事件
	keybord: false,
	//数据
	children:country
});
options.on('select',function(item){
	alert(JSON.stringify(item));
	this.hide();
})
$('#list-options').click(function() {
	options.show();
});

Event

监听list的事件 

var evt = new UIList({
	children:country
});
evt.on('shown',function(){
	alert('shown');
});
evt.on('hidden',function(){
	alert('hidden');
});
$('#list-event').click(function(){
	evt.show();
});

高级用法

通过renderItem，更改Item的渲染 

var list7 = new UIList({
	children:country,
	renderItem:function(item){
		return React.createElement('span',null,
			React.createElement('span',null,item.text),
			React.createElement('span',{className: item.value == 5 ? 'text-danger' : 'text-success'},'（'+item.value+'）')
		);
	}
});
$('#list7').click(function(){
	list7.show();
});
