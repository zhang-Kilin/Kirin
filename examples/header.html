Toolbar

工具栏，一行最多显示3个tool，超出的将自动转换为menu菜单，
目前已经定义的tool列表：home、earphone、share、edit、login、logout、cog、refresh、user、trash、barcode、qrcode、heart、search、favorite、print、envelope、plus、calendar 

var header3 = new UIHeader({
	title:'渲染到指定容器',
	render:'#header3',
	//定义工具栏
	toolbar:['home','share','search','favorite','print']
});
show code

Handles

toolbar的事件响应，事件响应规则：'on' + name 

var header4 = new UIHeader({
	title:'渲染到指定容器',
	render:'#header4',
	//定义工具栏
	toolbar:['home','share','search','favorite','print'],
	handles:{
		onhome:function(){
			alert('HOME');
		},
		onshare:function(){
			alert('SHARE');
		},
		onsearch:function(){
			alert('SEARCH');
		},
		onfavorite:function(){
			alert('FAVORITE');
		},
		onprint:function(){
			alert('PRINT');
		}
	}
});
show code

show

show方法支持动态更改Header的相关属性：title、toolbar、handles
方法定义如下：function(title,toolbar,handles){} 

var header6 = new UIHeader({
	title:'渲染到指定容器',
	render:'#header6',
	handles:{
		onback:function(){
			alert('BACK');
		}
	}
});

$('#header-show').click(function(){
	//动态更改header的属性
	header6.show('Hello world',['home'],{
		onback:function(){
			alert('hi back');
		},
		onhome:function(){
			alert('hi home');
		}
	});
});
