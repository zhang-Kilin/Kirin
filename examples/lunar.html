jsx.UILunarCalendar

jsx.UILunarCalendar原生react组件，为了便于扩展，未做Backbone.View的包装
API详见 jsx.UICalendar 

require(['jsx.UILunarCalendar','react','react-dom'],function(UILunarCalendar,React,ReactDOM){
	var calendar = ReactDOM.render(React.createElement(UILunarCalendar,{
		//当前要渲染的月份
		date:new Date(),
		//当前选中的日期，如果不需要默认选中日期，则currentDate=null即可
		currentDate:new Date
	},null),$('#calendar')[0]);
});
