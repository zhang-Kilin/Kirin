基本用法，带半透明遮罩，动画 

var normalAlert = new UIAlert();
$('#alert').click(function(){
	normalAlert.show("开关热i哦我附件的思考过");
});

Title

有标题栏的UIAlert，如果title为空，则不显示标题栏 

var titleAlert = new UIAlert();
$('#alert-title').click(function() {
	titleAlert.show("带Title的UIAlert",'UIAlert Title');
});
show code

Options

通过options参数配置alert 

var optionAlert = new UIAlert({
	//是否开启动画效果
	fade: false,
	//是否有半透明遮罩
	backdrop: false,
	//遮罩是否响应点击关闭事件
	keybord: false
});
$('#alert-options').click(function() {
	optionAlert.show("开关热i哦我附件的思考过");
});
show code

Callback

响应关闭后的回调函数，注意 如果有动画效果，则回调函数会在动画执行之前就被触发，如果需要监听动画完全执行完毕的事件，请参见 event 

var callbackAlert = new UIAlert();
$('#alert-callback').click(function() {
	callbackAlert.show("响应回调函数的用法",function(){
		alert('回调函数响应');
	});
});
show code

Event

监听UIAlert事件 

var eventAlert = new UIAlert();
eventAlert.on('hidden',function(){
	alert('hidden');
});
eventAlert.on('shown',function(){
	alert('shown');
});
$('#alert-event').click(function() {
	eventAlert.show("UIAlert事件监听",function(){
		alert('回调函数')
	});
});
show code

高级用法

定制化按钮、定制化内容 

//自定义按钮
var btnAlert = new UIAlert({
	buttons:[
		{
			text:'确定',
			handle:function(){
				btnAlert.hide();
			}
		}
	]
});
$('#alert-btn').click(function(){
	btnAlert.show('高级参数的UIAlert，自定义按钮');
});

//自定义内容UIAlert
var customAlert = new UIAlert();
$('#alert-custom').click(function(){
	var content = React.createElement('span',{className:'text-danger',style:{fontSize:'22px'}},'自定义内容UIAlert');
	customAlert.show(content);
});
